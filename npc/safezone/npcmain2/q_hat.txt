	Add(1,5277,1,500000,0,2211,1,903,50,7116,50,976,1);

	Add(1,5121,1,500000,0,7263,1,660,1,7099,30,7315,369);

	freeloop(0);
	set .menu$,"";
	for(set .@i,1; .@i<=getarraysize(.Shops$); set .@i,.@i+1) {
		set .menu$, .menu$+.Shops$[.@i]+":";
		npcshopdelitem "qitemshopnew"+.@i,909;
	}
	end;
	
    // เพิ่มส่วนนี้เพื่อให้ร้านค้าแสดงไอคอน
    for(set .@i,1; .@i<=getarraysize(.Shops$); set .@i,.@i+1) {
        npcshopitem "qitemshopnew"+.@i,0,0; // รีเซ็ตรายการไอเท็ม
    }
    end;

OnMenu:
	set .@size, getarraysize(@i);
	if (!.@size) set .@i, select(.menu$);
	else if (.@size == 1) set .@i, @i[0];
	else {
		for(set .@j,0; .@j<.@size; set .@j,.@j+1)
			set .@menu$, .@menu$+.Shops$[@i[.@j]]+":";
		set .@i, @i[select(.@menu$)-1];
	}
	deletearray @i[0],getarraysize(@i);
	if (.Shops$[.@i] == "") {
		message strcharinfo(0),"เกิดข้อผิดพลาดไม่ทราบสาเหตุ.";
		end;

	dispbottom "[ Quest Shop ] โปรดเลือกไอเท็มที่คุณต้องการทำเควส.",0xFFFFFF;
	callshop "qitemshopnew"+.@i,1;
	npcshopattach "qitemshopnew"+.@i;
	end;

OnBuyItem:
	setarray .@q[0],@bought_nameid[0],((@bought_quantity[0] > .MaxStack)?.MaxStack:@bought_quantity[0]);
	copyarray .@q[3],getd(".q_"+.@q[0]+"[0]"),getarraysize(getd(".q_"+.@q[0]));
	set .@q[2],.@q[1]*.@q[3];
	
	if (!.@q[2] || .@q[2] > 30000) {
		message strcharinfo(0),"คุณไม่สามารถซื้อ "+getitemname(.@q[0])+" ได้มากกว่า 30,000 รายการ.";
		end;
	}

	// ==========================================
	// คำนวณระบบตามที่ตั้งค่าไว้ใน OnInit
	// ==========================================
	set .@success_count, getd("$CraftCount_" + .@q[0]); 
	set .@base_zeny, .@q[4] * .@q[1];                   
	set .@current_zeny_cost, .@base_zeny + ((.@base_zeny * .Craft_Price_Inc * .@success_count) / 100); 
	
	set .@total_coin, .Craft_Coin_Amt * .@q[1]; 
	// ==========================================
	
	mes "[ ระบบสร้างไอเทมพิเศษ ]";
	mes "ไอเทมเป้าหมาย: ^0055FF" + ((.@q[2] > 1)?.@q[2]+"x ":"") + "^i["+.@q[0]+"]^FFFFFF " + getitemname(.@q[0]) + "^000000";
	mes "--------------------------------";
	mes "สิ่งที่ต้องใช้ในการสุ่มคราฟ:";
	
	if (.@base_zeny) {
		mes "~ ^FF9900Zeny: " + .@current_zeny_cost + " z ^777777(มี " + Zeny + " z)^000000";
		if (.@success_count > 0) {
			mes "  ^777777*ราคาบวกเพิ่ม " + (.@success_count * .Craft_Price_Inc) + "% (เซิร์ฟนี้คราฟติดไปแล้ว " + .@success_count + " ใบ)^000000";
		}
	}
	
	mes "~ ^i["+.Craft_Coin_ID+"]^FFFFFF " + getitemname(.Craft_Coin_ID) + " ^0066FF" + countitem(.Craft_Coin_ID) + " / " + .@total_coin + " ea^000000";
	
	mes "--------------------------------";
	mes "^FF0000[!] อัตราความสำเร็จ: " + .Craft_Success_Rate + "%^000000";
	mes "^FF0000[!] หากล้มเหลว ของและ Zeny จะสูญหาย^000000";
	mes "^FF0000[!] ไอเทมที่ได้จะสลักชื่อและผูกมัดติดไอดี^000000";
	mes "--------------------------------";
	
	next;
	setarray @qe[1], getiteminfo(.@q[0], 5), getiteminfo(.@q[0], 11);
	if (@qe[2] > 0 && ((@qe[1] & EQP_HEAD_LOW) || (@qe[1] & EQP_HEAD_TOP) || (@qe[1] & EQP_HEAD_MID) || (@qe[1] & EQP_COSTUME_HEAD_TOP) || (@qe[1] & EQP_COSTUME_HEAD_MID) || (@qe[1] & EQP_COSTUME_HEAD_LOW) || (@qe[1] & EQP_GARMENT) || (@qe[1] & EQP_COSTUME_GARMENT)))
		set .@preview,1;
	addtimer 1000, strnpcinfo(0)+"::OnEnd";
	
	while(1) {
		switch(select(" ~ ยืนยันการคราฟ ^FF0000("+.Craft_Success_Rate+"%)^000000:" + ((.@preview && !@qe[7])?" ~ ทดลองสวมใส่...:": "") + " ~ ยกเลิก")) {
		case 1:
			if (!checkweight(.@q[0],.@q[2])) {
				mes "[ ระบบสร้างไอเทมพิเศษ ]";
				mes "^FF0000น้ำหนักเกิน กรุณาเคลียร์ช่องเก็บของ^000000";
				close;
			}

			if (.@base_zeny && Zeny < .@current_zeny_cost) {
				mes "[ ระบบสร้างไอเทมพิเศษ ]";
				mes "^FF0000Zeny ไม่เพียงพอ (ต้องการ " + .@current_zeny_cost + " Zeny)^000000";
				close;
			}
			
			if (countitem(.Craft_Coin_ID) < .@total_coin) {
				mes "[ ระบบสร้างไอเทมพิเศษ ]";
				mes "^FF0000คุณมี " + getitemname(.Craft_Coin_ID) + " ไม่เพียงพอ^000000";
				close;
			}

			if (.@base_zeny) Zeny -= .@current_zeny_cost;
			delitem .Craft_Coin_ID, .@total_coin;

			if (rand(1, 100) <= .Craft_Success_Rate) { 
				
				set .@charid, getcharid(0); 
				setd "$CraftCount_" + .@q[0], .@success_count + 1;
				
				// สลักชื่อ + Account Bound 
				getitembound2 .@q[0], .@q[2], 1, 0, 0, 255, 0, (.@charid & 0xFFFF), (.@charid >> 16), 1;
				
				if (.Announce) announce "ยินดีด้วย! " + strcharinfo(0) + " สร้าง " + getitemname(.@q[0]) + " สำเร็จ!", 0;
				specialeffect2 EF_FLOWERLEAF;
				
				mes "[ ระบบสร้างไอเทมพิเศษ ]";
				mes "^009933คราฟสำเร็จ! คุณได้รับไอเทมแล้ว (สลักชื่อและติดไอดี)^000000";
				mes "สามารถนำหมวกไปให้อีก NPC สุ่มออฟชั่นได้เลย!";
			} else {
				specialeffect2 EF_BOMB;
				mes "[ ระบบสร้างไอเทมพิเศษ ]";
				mes "^FF0000คราฟล้มเหลว... วัตถุดิบและเงินสูญหาย^000000";
			}
			close;
			
		case 2:
			if (.@preview && !@qe[7]) {
				setarray @qe[3], getlook(LOOK_HEAD_BOTTOM), getlook(LOOK_HEAD_TOP), getlook(LOOK_HEAD_MID), getlook(LOOK_ROBE), 1;
				if ((@qe[1] & 1) || (@qe[1] & 4096)) changelook LOOK_HEAD_BOTTOM, @qe[2];
				else if ((@qe[1] & 256) || (@qe[1] & 1024)) changelook LOOK_HEAD_TOP, @qe[2];
				else if ((@qe[1] & 512) || (@qe[1] & 2048)) changelook LOOK_HEAD_MID, @qe[2];
				else if ((@qe[1] & 4) || (@qe[1] & 8192)) changelook LOOK_ROBE, @qe[2];
				break;
			} else {
				close;
			}
		case 3:
			close;
		}
	}

OnEnd:
	if (@qe[7]) {
		changelook LOOK_HEAD_BOTTOM, @qe[3];
		changelook LOOK_HEAD_TOP, @qe[4];
		changelook LOOK_HEAD_MID, @qe[5];
		changelook LOOK_ROBE, @qe[6];
	}
	deletearray @qe[0],8;
	end;

function Add {
	if (getitemname(getarg(1)) == "null") {
		debugmes "Quest reward #"+getarg(1)+" invalid (skipped).";
		return;
	}
	
	setarray .@j[0],getarg(2),getarg(3),getarg(4);
	copyarray getd(".q_"+getarg(1)+"[0]"),.@j[0],getarraysize(.@j);
	npcshopadditem "qitemshopnew"+getarg(0),getarg(1),((.ShowZeny)?getarg(3):0);
	return;
}

function Chk {
	if (getarg(0) < getarg(1)) {
		set @qe[0],1;
		return "^FF0000";
	} else
		return "^009933";
}

function Slot {
    set .@s$,"^i[" + getarg(0) + "]^FFFFFF " + getitemname(getarg(0)) + "^000000";
    switch(.ShowSlot) {
        case 1: if (!getitemslots(getarg(0))) return .@s$;
        case 2: if (getiteminfo(getarg(0), 2) == 4 || getiteminfo(getarg(0), 2) == 5) return .@s$ + " ["+getitemslots(getarg(0))+"]";
        default: return .@s$;
    }
}

function A_An {
	setarray .@A$[0],"a","e","i","o","u";
	set .@B$, "_"+getarg(0);
	for(set .@i,0; .@i<5; set .@i,.@i+1)
		if (compare(.@B$,"_"+.@A$[.@i])) return " "+getarg(0);
	return " "+getarg(0);
}
}

	deletearray @i[0],getarraysize(@i);
	for(set .@i,0; .@i<getargcount(); set .@i,.@i+1)
		set @i[.@i],getarg(.@i);
	doevent "questi_shop1::OnMenu";
	end;
}
