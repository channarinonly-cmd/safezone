function	script	finalskill	{
	if (finalskill == 10) {
		dispbottom "ไม่สามารถใช้งานได้อีกต่อไป (เลเวลสกิลสูงสุดแล้ว)",0xFFFFFF;
		end;
	}
	if (countitem(3600) < 2) {
		dispbottom "ไม่สามารถใช้งานได้ (ต้องการ Final Skill Card 2 ea)",0xFFFFFF;
		end;
	}
	set .@add_carry,finalskill + 1;
	set finalskill,.@add_carry;
	skill "KN_BOWLINGBASH",.@add_carry,SKILL_PERM_GRANT;
	delitem 3600,2;
	specialeffect2 EF_EARTHSPIKE;
	dispbottom "คุณทำการใช้ Final Skill Card ทั้งหมด 2 ea เพื่อเพิ่มน้ำหนักสำเร็จ",0x000000;
	end;
}
