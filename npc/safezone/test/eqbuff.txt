function	script	F_IBUFF_PosToIdx	{
	.@pos = getarg(0);
	if (.@pos & EQP_ARMOR)    return 0;
	if (.@pos & EQP_HEAD_TOP) return 1;
	if (.@pos & EQP_HAND_L)   return 2;
	if (.@pos & EQP_SHOES)    return 3;
	if (.@pos & EQP_GARMENT)  return 4;
	if (.@pos & EQP_HAND_R)   return 5;
	if (.@pos & EQP_WEAPON)   return 5;
	return -1;
}

function	script	F_IBUFF_SlotToIdx	{
	.@s = getarg(0);
	if (.@s == EQI_ARMOR) return 0; if (.@s == EQI_HEAD_TOP) return 1;
	if (.@s == EQI_HAND_L) return 2; if (.@s == EQI_SHOES) return 3;
	if (.@s == EQI_GARMENT) return 4; if (.@s == EQI_HAND_R) return 5;
	return -1;
}

function	script	F_IBUFF_IdxToSlot	{
	.@idx = getarg(0);
	switch (.@idx) {
		case 0: return EQI_ARMOR; case 1: return EQI_HEAD_TOP; case 2: return EQI_HAND_L;
		case 3: return EQI_SHOES; case 4: return EQI_GARMENT; case 5: return EQI_HAND_R;
	}
	return -1;
}

function	script	F_IBUFF_SlotName$	{
	.@idx = getarg(0);
	switch (.@idx) {
		case 0: return "Armor"; case 1: return "Headgear"; case 2: return "Shield";
		case 3: return "Footgear"; case 4: return "Garment"; case 5: return "Weapon";
	}
	return "Unknown";
}

function	script	F_IBUFF_FormatName$	{
	.@s$ = getarg(0);
	.@s$ = replacestr(.@s$, "{", ""); .@s$ = replacestr(.@s$, "}", "");
	.@s$ = replacestr(.@s$, "\t", " "); .@s$ = replacestr(.@s$, ";", "; ");
	.@s$ = replacestr(.@s$, "  ", " ");
	.@s$ = replacestr(.@s$, "  ", " ");
	while (getstrlen(.@s$) > 0 && substr(.@s$, 0, 1) == " ") { .@len = getstrlen(.@s$);
		if (.@len <= 1) .@s$ = ""; else .@s$ = substr(.@s$, 1, .@len - 1);
	}
	while (getstrlen(.@s$) > 0 && substr(.@s$, getstrlen(.@s$) - 1, 1) == " ") { .@len = getstrlen(.@s$);
		if (.@len <= 1) .@s$ = ""; else .@s$ = substr(.@s$, 0, .@len - 1);
	}
	if (getstrlen(.@s$) > 90) .@s$ = substr(.@s$, 0, 90) + "...";
	return .@s$;
}

function	script	F_IBUFF_NormalizeScript$	{
	.@s$ = getarg(0); .@len = getstrlen(.@s$);
	if (.@len <= 0) return "{ }";
	if (substr(.@s$, 0, 1) != "{") .@s$ = "{ " + .@s$;
	.@len = getstrlen(.@s$);
	if (.@len > 0 && substr(.@s$, .@len-1, 1) != "}") .@s$ = .@s$ + " }";
	return .@s$;
}

function	script	F_IBUFF_EscapeSQL$	{
	.@s$ = getarg(0); .@s$ = replacestr(.@s$, "\\", "\\\\"); .@s$ = replacestr(.@s$, "'", "''"); return .@s$;
}

		.@eqid = getequipid(.@slot);
		
		if (.@eqid < 0) { 
			query_sql("DELETE FROM `item_buff_slot` WHERE `char_id`="+.@cid+" AND `slot_idx`="+.@i+" AND `is_locked`=0");
			continue; 
		}
		
		.@eqref = getequiprefinerycnt(.@slot);
		.@r = query_sql("SELECT `item_id`,`refine`,`script`,`expire_ts`,`is_locked` FROM `item_buff_slot` WHERE `char_id`="+.@cid+" AND `slot_idx`="+.@i+" LIMIT 1", .@db_iid, .@db_ref, .@scr$, .@exp, .@locked);
		if (.@r <= 0) continue;
		
		if (.@db_iid != .@eqid || .@db_ref != .@eqref) { 
			if (.@locked == 0) { query_sql("DELETE FROM `item_buff_slot` WHERE `char_id`="+.@cid+" AND `slot_idx`="+.@i); }
			continue; 
		}
		
		.@rem = .@exp - .@now;
		if (.@rem <= 0) { query_sql("DELETE FROM `item_buff_slot` WHERE `char_id`="+.@cid+" AND `slot_idx`="+.@i); continue; }
		
		// กำหนดไอคอนให้แสดงผลถูกต้อง (ใช้ EFST_ แทน SC_)
		.@icon = EFST_BLESSING;
		if (.@i == 0) .@icon = EFST_DEFENDER;
		else if (.@i == 1) .@icon = EFST_MAGNIFICAT;
		else if (.@i == 2) .@icon = EFST_REFLECTSHIELD;
		else if (.@i == 3) .@icon = EFST_WINDWALK;
		else if (.@i == 4) .@icon = EFST_ASSUMPTIO;
		else if (.@i == 5) {
			// ตั้งค่าเริ่มต้นเป็น 0 (ไม่แสดงรูปไอคอนใดๆ)
			.@icon = 0; 
			
			// เช็คธาตุ ถ้าพบว่ามีโบนัสธาตุ ถึงจะแสดงรูปไอคอน
			if (replacestr(.@scr$, "bAtkEle,3", "") != .@scr$) .@icon = EFST_PROPERTYFIRE;
			else if (replacestr(.@scr$, "bAtkEle,1", "") != .@scr$) .@icon = EFST_PROPERTYWATER;
			else if (replacestr(.@scr$, "bAtkEle,4", "") != .@scr$) .@icon = EFST_PROPERTYWIND;
			else if (replacestr(.@scr$, "bAtkEle,2", "") != .@scr$) .@icon = EFST_PROPERTYGROUND;
			else if (replacestr(.@scr$, "bAtkEle,6", "") != .@scr$) .@icon = EFST_ASPERSIO;
		}

		// แปะบัฟ (เวลาใส่เป็น .@rem วินาทีเพียวๆ ไม่ต้องคูณ 1000 แล้วครับ)
		.@tmp$ = callfunc("F_IBUFF_NormalizeScript$", .@scr$);
		bonus_script .@tmp$, .@rem, 2, 1, .@icon;
		
		if (.@next_exp == 0 || .@exp < .@next_exp) { .@next_exp = .@exp; .@next_idx = .@i; }
	}
	
	if (.@next_exp > 0) {
		.@sec = .@next_exp - .@now; if (.@sec < 1) .@sec = 1;
		@IBUFF_WARN_IDX = .@next_idx;
		if (.@sec > 600) addtimer ((.@sec - 600) * 1000), "ItemBuffSystem::OnIBUFF_Warn10Min";
		if (.@sec > 60) addtimer ((.@sec - 60) * 1000), "ItemBuffSystem::OnIBUFF_Warn1Min";
		addtimer (.@sec * 1000) + 50, "ItemBuffSystem::OnIBUFF_ExpireTick";
	} else { 
		@IBUFF_WARN_IDX = -1; 
	}
	return;
}
