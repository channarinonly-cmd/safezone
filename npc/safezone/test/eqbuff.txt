//======================================================
// Item Buff Master (v7 - Renewal Safe + Select Style)
// -----------------------------------------------------
// - Renewal-safe: ไม่ต้องระบุ item ID whitelist แล้ว ใส่อยู่ + refine ถึงขั้นต่ำก็สุ่มได้
// - ของสวมใส่ใช้ pool รวม และผู้เล่นเลือกสายก่อนสุ่ม
// - อาวุธใช้ pool อาวุธแยกจากของสวมใส่ แต่ไม่ล็อกตาม item ID
// - ไม่มีสุ่มธาตุ ให้ผู้เล่นใช้ใบธาตุ/สกิลเคลือบเอง
// - Ignore DEF อยู่เฉพาะสายอาวุธกายภาพ และ cap สูงสุด 20%
// - Renewal balance: ลด % ที่ stack ง่าย, เพิ่ม HIT/FLEE แบบ flat ให้รู้สึกมีผล
//======================================================

morocc,151,54,5	script	Item Buff Master::Item_Buff_NPC	4_F_KAFRA1,{

	if (strcharinfo(3) == "morocc") {
		callfunc "Check_Faction", 1;
	} else if (strcharinfo(3) == "prontera") {
		callfunc "Check_Faction", 2; 
	}

	.cost = 10000;
	.buff_sec = 4 * 60 * 60; // 4 ชั่วโมง
	.min_ref = 6; // refine ขั้นต่ำที่รับบัฟได้
	.lock_cost = 1; 
	.debug = 0; 

	// v7: ไม่มี item whitelist แล้ว ทุกอุปกรณ์ในช่องที่รองรับใช้ได้เมื่อ refine ถึง .min_ref

	mes "[Item Buff Master]";
	mes "ไอเทมของคุณน่าเบื่ออยู่หรือเปล่า?? !";
	mes "มาคุยกับข้า อุปกรณ์ที่ตีบวกถึงขั้นต่ำจะสุ่มบัฟได้!";
	next;
	
	mes "[Item Buff Master]";
	mes "^FF0000เมื่อเสีย 10,000z จะสุ่มรับพลังแฝง 1 อย่าง และจะอยู่ 4 ชั่วโมง^000000";
	mes "^FF0000ถ้าถอด/สลับไอเทมในสล็อตนั้น บัฟจะหายทันที^000000";
	mes "^00AAFF(ยกเว้นท่านจะใช้บริการ ผนึกบัฟ กับข้า)^000000";
	next;
	mes "[Item Buff Master]";
	mes "เลือกเมนู:";
	next;

	switch(select("สุ่มบัฟ:ดูรายการบัฟที่สุ่มได้ (Preview):ดูผังบัฟรวม/เลือกสาย:ดูเงื่อนไข/ผังบัฟ:ดูบัฟที่ใช้อยู่ (สรุป):ผนึกบัฟกันสูญหาย (Lock Buff)")) {
		case 2: callsub S_PreviewMenu; close;
		case 3: callsub S_BuffGuideMenu; close;
		case 4: callsub S_ShowAllowedItems; close;
		case 5: callsub S_ShowAllBuffs_Summary; close;
		case 6: callsub S_LockBuffMenu; close;
		case 1: break;
	}

	callsub S_ShowAllBuffs;
	if (Zeny < .cost) { mes "เงินเจ้าไม่พอ!"; close; }

	mes "[Item Buff Master]";
	mes "เลือกชนิดอุปกรณ์ที่ต้องการสุ่มบัฟ";
	next;

	.@slot = callsub(S_SelectEquipSlot);
	if (.@slot < 0) { mes "เจ้าต้องสวมใส่อุปกรณ์ก่อน!"; close; }

	.@iid = getequipid(.@slot);
	.@ref = getequiprefinerycnt(.@slot);
	if (.@ref < .min_ref) { mes "[Item Buff Master]"; mes "ต้องเป็นไอเทมตีบวก +"+.min_ref+" ขึ้นไป!"; close; }

	.@groupID = callsub(S_GetItemGroup, .@slot, .@iid);
	if (.@groupID == 0) { mes "[Item Buff Master]"; mes "ช่องนี้ยังไม่รองรับระบบบัฟ..."; close; }

	.@idx = callfunc("F_IBUFF_SlotToIdx", .@slot);

	.@style = callsub(S_SelectBuffStyle, .@slot, .@groupID);
	if (.@style <= 0) close;

	mes "[Item Buff Master]";
	mes "ช่อง: ^0000FF"+callfunc("F_IBUFF_SlotName$", .@idx)+"^000000";
	mes "สายที่เลือก: ^00AAFF"+callfunc("F_IBUFF_StyleName$", .@style)+"^000000";
	mes "เมื่อสุ่มใหม่ บัฟเดิมของช่องนี้จะหายทันที";
	mes "ต้องการสุ่มหรือไม่?";
	next;
	if (select("สุ่มบัฟ:ยกเลิก") == 2) close;
	
	if (getequipid(.@slot) != .@iid || getequiprefinerycnt(.@slot) != .@ref) { mes "[Item Buff Master]"; mes "อุปกรณ์มีการเปลี่ยนแปลง ข้าทำพิธีต่อไม่ได้!"; close; }
	if (Zeny < .cost) { mes "[Item Buff Master]"; mes "อ้าว! เงินเจ้าน่ะไม่พอแล้ว!"; close; }

	Zeny -= .cost;

	@IBUFF_SLOT = .@slot; @IBUFF_IID = .@iid; @IBUFF_REF = .@ref; @IBUFF_IDX = .@idx; @IBUFF_NPCDBG = .debug;
	if (.@ref >= 9) .@tier = 9;
	else if (.@ref >= 8) .@tier = 8;
	else if (.@ref >= 7) .@tier = 7;
	else .@tier = 6;

	callsub S_RollAndApplyByGroup, .@groupID, .@tier, .@style;

	mes "[Item Buff Master]";
	mes "บัฟถูกติดตั้งเรียบร้อย! (4 ชั่วโมง)";
	next;
	callsub S_ShowAllBuffs;
	close;

S_BuffGuideMenu:
	mes "[Item Buff Master]";
	mes "v7 Renewal เปลี่ยนแนวคิดใหม่:";
	mes "^00AAFFไม่ต้องระบุ item ID whitelist แล้ว^000000";
	mes "อุปกรณ์ทุกชิ้นในช่องที่รองรับ ถ้าตีบวกถึงขั้นต่ำก็สุ่มบัฟได้";
	mes "เงื่อนไขตอนนี้คือ: ^FF0000สวมอยู่ + refine +"+.min_ref+" ขึ้นไป^000000";
	next;
	mes "[ของสวมใส่: Armor/Head/Shield/Shoes/Garment]";
	callsub S_PrintWearPool;
	next;
	mes "[อาวุธ: Weapon]";
	callsub S_PrintWepPool;
	close;

S_PrintWearPool:
	mes "ของสวมใส่ใช้ pool รวม และเลือกสายก่อนสุ่มได้ 5 สาย:";
	mes " 1) กายภาพ - ATK/HIT/ASPD/HP";
	mes " 2) คริ/AGI - Critical/ASPD/FLEE/HIT";
	mes " 3) เวท/ซัพ - MATK/Cast/Delay/SP";
	mes " 4) ระยะไกล - Long Range/HIT/ASPD/Critical";
	mes " 5) อึด-SP - MaxHP/MaxSP/ลดใช้ SP/SP Recovery";
	mes "^888888หมายเหตุ: ของสวมใส่ไม่มี Ignore DEF เพื่อกัน stack ทะลุ 20%^000000";
	return;

S_PrintWepPool:
	mes "อาวุธไม่ล็อก item ID แล้ว ทุกอาวุธ +"+.min_ref+" ขึ้นไปใช้ได้";
	mes "เลือกสายก่อนสุ่มได้ 5 สาย:";
	mes " 1) กายภาพ - ATK/HIT/Ignore DEF/HP";
	mes " 2) คริ/AGI - Critical/ASPD/FLEE/HIT";
	mes " 3) เวท/ซัพ - MATK/Cast/Delay/SP";
	mes " 4) ระยะไกล - Long Range/HIT/ASPD/Critical";
	mes " 5) อึด-SP - MaxHP/MaxSP/ลดใช้ SP/SP Recovery";
	mes "^FF0000Renewal: ลด % ที่ stack ง่าย และ Ignore DEF cap สูงสุด 20%^000000";
	mes "ไม่มีสุ่มธาตุ ให้ผู้เล่นใช้ใบธาตุ/สกิลเคลือบเอง";
	return;

S_PrintGroupOptions:
	.@grp = getarg(0);
	if (.@grp >= 61) {
		mes "  Pool อาวุธ:";
		mes "   - กายภาพ / คริ-AGI / เวท-ซัพ / ระยะไกล / อึด-SP";
		mes "   - สายกายภาพอาวุธมีโอกาสได้ Ignore DEF สูงสุด 20%";
		return;
	}
	mes "  Pool ของสวมใส่:";
	mes "   - กายภาพ / คริ-AGI / เวท-ซัพ / ระยะไกล / อึด-SP";
	mes "   - เลือกสายก่อนสุ่ม ไม่แยกตาม item ID แล้ว";
	return;

S_ShowAllowedItems:
	mes "[Item Buff Master]";
	mes "v7 ไม่มีรายชื่อ เงื่อนไข slot แล้ว";
	mes "อุปกรณ์ที่รองรับมี 6 ช่อง:";
	mes " - Armor";
	mes " - Headgear ส่วนบน";
	mes " - Shield / มือซ้าย";
	mes " - Footgear";
	mes " - Garment";
	mes " - Weapon / มือขวา";
	mes "--------------------------------";
	mes "เงื่อนไข: ต้องสวมอยู่ และ refine +"+.min_ref+" ขึ้นไป";
	mes "ถอด/เปลี่ยน item ID หรือ refine ไม่ตรง บัฟจะไม่ทำงาน";
	next;
	mes "[ของสวมใส่]";
	callsub S_PrintWearPool;
	next;
	mes "[อาวุธ]";
	callsub S_PrintWepPool;
	close;

S_GetItemGroup:
	.@slot = getarg(0); .@iid = getarg(1);
	// v7: ไม่เช็ค item ID แล้ว ใช้ slot เป็นตัวกำหนดประเภทเท่านั้น
	if (.@slot == EQI_ARMOR) return 10;
	if (.@slot == EQI_HEAD_TOP) return 20;
	if (.@slot == EQI_HAND_L) return 30;
	if (.@slot == EQI_SHOES) return 40;
	if (.@slot == EQI_GARMENT) return 50;
	if (.@slot == EQI_HAND_R) return 61;
	return 0;

S_PreviewMenu:
	mes "[Item Buff Master]";
	mes "เลือกอุปกรณ์ที่สวมใส่อยู่เพื่อดูรายการบัฟที่เป็นไปได้";
	next;
	.@slot = callsub(S_SelectEquipSlot);
	if (.@slot < 0) { mes "ท่านยังไม่ได้สวมใส่อุปกรณ์ใดๆ!"; close; }
	.@iid = getequipid(.@slot); .@ref = getequiprefinerycnt(.@slot);
	if (.@ref < .min_ref) { mes "[Item Buff Master]"; mes "ไอเทมต้องตีบวก +"+.min_ref+" ขึ้นไป จึงจะมีบัฟแฝงให้ท่านสุ่มได้นะ!"; close; }
	.@groupID = callsub(S_GetItemGroup, .@slot, .@iid);
	if (.@groupID == 0) { mes "[Item Buff Master]"; mes "ช่องนี้ยังไม่รองรับระบบบัฟ..."; close; }

	if (.@ref >= 9) .@tier = 9; else if (.@ref >= 8) .@tier = 8; else if (.@ref >= 7) .@tier = 7; else .@tier = 6;
	mes "[Item Buff Master]";
	mes "ไอเทม: ^FF0000" + getitemname(.@iid) + " +" + .@ref + "^000000";
	mes "ระดับขั้นบัฟ: ^00AAFFTier " + .@tier + "^000000";
	mes "ประเภทช่อง: ^00AAFF" + callfunc("F_IBUFF_GroupTitle$", .@groupID) + "^000000";
	mes "^888888" + callfunc("F_IBUFF_GroupDesc$", .@groupID) + "^000000";
	mes "--------------------------------";
	callsub S_PrintGroupOptions, .@groupID;
	next;
	callsub S_PrintTierPrevOpt, .@groupID, .@tier;
	close;

S_PrintTierPrevOpt:
	.@grp = getarg(0); .@tier = getarg(1);
	.@drop = callsub(S_TierVal, .@tier, 1, 2, 3, 5);
	.@aspd = callsub(S_TierVal, .@tier, 3, 5, 7, 10);
	.@crit = callsub(S_TierVal, .@tier, 5, 8, 11, 15);
	.@atk = callsub(S_TierVal, .@tier, 3, 5, 7, 10);
	.@matk = callsub(S_TierVal, .@tier, 3, 5, 7, 10);
	.@cast = callsub(S_TierVal, .@tier, 3, 5, 7, 10);
	.@delay = callsub(S_TierVal, .@tier, 3, 5, 7, 10);
	.@long = callsub(S_TierVal, .@tier, 3, 5, 7, 10);
	.@hit = callsub(S_TierVal, .@tier, 10, 15, 20, 30);
	.@flee = callsub(S_TierVal, .@tier, 10, 15, 20, 30);
	.@sp = callsub(S_TierVal, .@tier, 5, 8, 12, 15);
	.@hp = callsub(S_TierVal, .@tier, 3, 5, 7, 10);
	.@ignore = callsub(S_TierVal, .@tier, 5, 10, 15, 20);
	mes "[ค่าประมาณของ Tier นี้]";
	mes "Drop +"+.@drop+"%, ASPD +"+.@aspd+"%, Critical +"+.@crit;
	mes "ATK/MATK +"+.@atk+"%, Cast/Delay -"+.@cast+"%";
	mes "Long Range +"+.@long+"%, HIT/FLEE +"+.@hit;
	mes "SP sustain " + .@sp + "%, HP/SP +"+.@hp+"%";
	if (.@grp >= 61) mes "Ignore DEF มีเฉพาะสายกายภาพอาวุธ สูงสุด ^FF0000"+.@ignore+"%^000000 และ cap ที่ 20%";
	mes "--------------------------------";
	mes "หมายเหตุ: รายละเอียดจริงจะขึ้นตาม pool ที่สุ่มได้";
	return;

S_TierVal:
	.@tier = getarg(0);
	if (.@tier >= 9) return getarg(4);
	if (.@tier >= 8) return getarg(3);
	if (.@tier >= 7) return getarg(2);
	return getarg(1);

S_SelectBuffStyle:
	.@slot = getarg(0); .@grp = getarg(1);
	mes "[Item Buff Master]";
	mes "เลือกสายบัฟที่ต้องการสุ่ม";
	if (.@grp >= 61) {
		mes "^00AAFFอาวุธใช้ pool อาวุธ ไม่ล็อกตาม item ID^000000";
		mes "สายกายภาพอาวุธมีโอกาสได้ Ignore DEF สูงสุด 20%";
	} else {
		mes "^00AAFFของสวมใส่ทุกช่องใช้ pool รวมเดียวกัน^000000";
		mes "ของสวมใส่ไม่มี Ignore DEF";
	}
	.@s = select("สายกายภาพ:สายคริ/AGI:สายเวท/ซัพ:สายระยะไกล:สายอึด-SP:ยกเลิก");
	if (.@s >= 1 && .@s <= 5) return .@s;
	return 0;

S_RollAndApplyByGroup:
	.@grp = getarg(0); .@tier = getarg(1);
	.@style = (getargcount() >= 3) ? getarg(2) : 1;
	.@drop = callsub(S_TierVal, .@tier, 1, 2, 3, 5);
	.@aspd = callsub(S_TierVal, .@tier, 3, 5, 7, 10);
	.@crit = callsub(S_TierVal, .@tier, 5, 8, 11, 15);
	.@atk = callsub(S_TierVal, .@tier, 3, 5, 7, 10);
	.@matk = callsub(S_TierVal, .@tier, 3, 5, 7, 10);
	.@cast = callsub(S_TierVal, .@tier, 3, 5, 7, 10);
	.@delay = callsub(S_TierVal, .@tier, 3, 5, 7, 10);
	.@long = callsub(S_TierVal, .@tier, 3, 5, 7, 10);
	.@hit = callsub(S_TierVal, .@tier, 10, 15, 20, 30);
	.@flee = callsub(S_TierVal, .@tier, 10, 15, 20, 30);
	.@sp = callsub(S_TierVal, .@tier, 5, 8, 12, 15);
	.@hp = callsub(S_TierVal, .@tier, 3, 5, 7, 10);
	.@ignore = callsub(S_TierVal, .@tier, 5, 10, 15, 20);
	if (.@ignore > 20) .@ignore = 20;

	if (.@grp >= 61) {
		callsub S_ApplyWepStyle, .@grp, .@style, .@drop, .@aspd, .@crit, .@atk, .@matk, .@cast, .@delay, .@long, .@hit, .@flee, .@sp, .@hp, .@ignore;
		return;
	}
	callsub S_ApplyEquipStyle, .@style, .@drop, .@aspd, .@crit, .@atk, .@matk, .@cast, .@delay, .@long, .@hit, .@flee, .@sp, .@hp;
	return;

S_ApplyEquipStyle:
	.@style=getarg(0); .@drop=getarg(1); .@aspd=getarg(2); .@crit=getarg(3); .@atk=getarg(4); .@matk=getarg(5); .@cast=getarg(6); .@delay=getarg(7); .@long=getarg(8); .@hit=getarg(9); .@flee=getarg(10); .@sp=getarg(11); .@hp=getarg(12);
	.@pick = rand(4);
	if (.@style == 1) {
		if (.@pick==0) callsub S_Apply, "{ bonus bAtkRate,"+.@atk+"; bonus bHit,"+.@hit+"; bonus2 bDropAddRace,RC_All,"+.@drop+"; }", " - ATK +"+.@atk+"%\n- HIT +"+.@hit+"\n- DROP +"+.@drop+"%";
		else if (.@pick==1) callsub S_Apply, "{ bonus bAspdRate,"+.@aspd+"; bonus bAtkRate,"+.@atk+"; bonus2 bDropAddRace,RC_All,"+.@drop+"; }", " - ASPD +"+.@aspd+"%\n- ATK +"+.@atk+"%\n- DROP +"+.@drop+"%";
		else if (.@pick==2) callsub S_Apply, "{ bonus bHit,"+.@hit+"; bonus bFlee,"+.@flee+"; bonus2 bDropAddRace,RC_All,"+.@drop+"; }", " - HIT +"+.@hit+"\n- FLEE +"+.@flee+"\n- DROP +"+.@drop+"%";
		else callsub S_Apply, "{ bonus bMaxHPrate,"+.@hp+"; bonus bAtkRate,"+.@atk+"; bonus2 bDropAddRace,RC_All,"+.@drop+"; }", " - MaxHP +"+.@hp+"%\n- ATK +"+.@atk+"%\n- DROP +"+.@drop+"%";
		return;
	}
	if (.@style == 2) {
		if (.@pick==0) callsub S_Apply, "{ bonus bCritical,"+.@crit+"; bonus bAspdRate,"+.@aspd+"; bonus2 bDropAddRace,RC_All,"+.@drop+"; }", " - Critical +"+.@crit+"\n- ASPD +"+.@aspd+"%\n- DROP +"+.@drop+"%";
		else if (.@pick==1) callsub S_Apply, "{ bonus bCritical,"+.@crit+"; bonus bFlee,"+.@flee+"; bonus2 bDropAddRace,RC_All,"+.@drop+"; }", " - Critical +"+.@crit+"\n- FLEE +"+.@flee+"\n- DROP +"+.@drop+"%";
		else if (.@pick==2) callsub S_Apply, "{ bonus bCritical,"+.@crit+"; bonus bHit,"+.@hit+"; bonus2 bDropAddRace,RC_All,"+.@drop+"; }", " - Critical +"+.@crit+"\n- HIT +"+.@hit+"\n- DROP +"+.@drop+"%";
		else callsub S_Apply, "{ bonus bAspdRate,"+.@aspd+"; bonus bFlee,"+.@flee+"; bonus2 bDropAddRace,RC_All,"+.@drop+"; }", " - ASPD +"+.@aspd+"%\n- FLEE +"+.@flee+"\n- DROP +"+.@drop+"%";
		return;
	}
	if (.@style == 3) {
		if (.@pick==0) callsub S_Apply, "{ bonus bMatkRate,"+.@matk+"; bonus bCastrate,-"+.@cast+"; bonus2 bDropAddRace,RC_All,"+.@drop+"; }", " - MATK +"+.@matk+"%\n- ลด Cast "+.@cast+"%\n- DROP +"+.@drop+"%";
		else if (.@pick==1) callsub S_Apply, "{ bonus bMatkRate,"+.@matk+"; bonus bDelayrate,-"+.@delay+"; bonus2 bDropAddRace,RC_All,"+.@drop+"; }", " - MATK +"+.@matk+"%\n- ลด Delay "+.@delay+"%\n- DROP +"+.@drop+"%";
		else if (.@pick==2) callsub S_Apply, "{ bonus bSPrecovRate,"+.@sp+"; bonus bMatkRate,"+.@matk+"; bonus2 bDropAddRace,RC_All,"+.@drop+"; }", " - SP Recovery +"+.@sp+"%\n- MATK +"+.@matk+"%\n- DROP +"+.@drop+"%";
		else callsub S_Apply, "{ bonus bUseSPrate,-"+.@sp+"; bonus bMaxSPrate,"+.@hp+"; bonus2 bDropAddRace,RC_All,"+.@drop+"; }", " - ลดใช้ SP "+.@sp+"%\n- MaxSP +"+.@hp+"%\n- DROP +"+.@drop+"%";
		return;
	}
	if (.@style == 4) {
		if (.@pick==0) callsub S_Apply, "{ bonus bLongAtkRate,"+.@long+"; bonus bHit,"+.@hit+"; bonus2 bDropAddRace,RC_All,"+.@drop+"; }", " - Long Range +"+.@long+"%\n- HIT +"+.@hit+"\n- DROP +"+.@drop+"%";
		else if (.@pick==1) callsub S_Apply, "{ bonus bLongAtkRate,"+.@long+"; bonus bAspdRate,"+.@aspd+"; bonus2 bDropAddRace,RC_All,"+.@drop+"; }", " - Long Range +"+.@long+"%\n- ASPD +"+.@aspd+"%\n- DROP +"+.@drop+"%";
		else if (.@pick==2) callsub S_Apply, "{ bonus bLongAtkRate,"+.@long+"; bonus bCritical,"+.@crit+"; bonus2 bDropAddRace,RC_All,"+.@drop+"; }", " - Long Range +"+.@long+"%\n- Critical +"+.@crit+"\n- DROP +"+.@drop+"%";
		else callsub S_Apply, "{ bonus bHit,"+.@hit+"; bonus bFlee,"+.@flee+"; bonus2 bDropAddRace,RC_All,"+.@drop+"; }", " - HIT +"+.@hit+"\n- FLEE +"+.@flee+"\n- DROP +"+.@drop+"%";
		return;
	}

function	script	F_IBUFF_PosToIdx	{
	.@pos = getarg(0);
	if (.@pos & EQP_ARMOR)    return 0;
	if (.@pos & EQP_HEAD_TOP) return 1;
	if (.@pos & EQP_HAND_L)   return 2;
	if (.@pos & EQP_SHOES)    return 3;
	if (.@pos & EQP_GARMENT)  return 4;
	if (.@pos & EQP_HAND_R)   return 5;
	if (.@pos & EQP_WEAPON)   return 5;
	return -1;
}

function	script	F_IBUFF_SlotToIdx	{
	.@s = getarg(0);
	if (.@s == EQI_ARMOR) return 0; if (.@s == EQI_HEAD_TOP) return 1;
	if (.@s == EQI_HAND_L) return 2; if (.@s == EQI_SHOES) return 3;
	if (.@s == EQI_GARMENT) return 4; if (.@s == EQI_HAND_R) return 5;
	return -1;
}

function	script	F_IBUFF_IdxToSlot	{

function	script	F_IBUFF_FormatName$	{
	.@s$ = getarg(0);
	.@s$ = replacestr(.@s$, "{", ""); .@s$ = replacestr(.@s$, "}", "");
	.@s$ = replacestr(.@s$, "\t", " "); .@s$ = replacestr(.@s$, ";", "; ");
	.@s$ = replacestr(.@s$, "  ", " ");
	.@s$ = replacestr(.@s$, "  ", " ");
	while (getstrlen(.@s$) > 0 && substr(.@s$, 0, 1) == " ") { .@len = getstrlen(.@s$);
		if (.@len <= 1) .@s$ = ""; else .@s$ = substr(.@s$, 1, .@len - 1);
	}
	while (getstrlen(.@s$) > 0 && substr(.@s$, getstrlen(.@s$) - 1, 1) == " ") { .@len = getstrlen(.@s$);
		if (.@len <= 1) .@s$ = ""; else .@s$ = substr(.@s$, 0, .@len - 1);
	}
	if (getstrlen(.@s$) > 90) .@s$ = substr(.@s$, 0, 90) + "...";
	return .@s$;
}

function	script	F_IBUFF_NormalizeScript$	{
	.@s$ = getarg(0); .@len = getstrlen(.@s$);
	if (.@len <= 0) return "{ }";
	// Safety cap: record เก่าที่เคยสุ่ม Ignore DEF 30/40 จะถูกบังคับลงมาไม่เกิน 20 ตอน apply
	.@s$ = replacestr(.@s$, "bonus2 bIgnoreDefRaceRate,RC_All,40", "bonus2 bIgnoreDefRaceRate,RC_All,20");
	.@s$ = replacestr(.@s$, "bonus2 bIgnoreDefRaceRate,RC_All,30", "bonus2 bIgnoreDefRaceRate,RC_All,20");
	// v4: กัน record เก่าที่มีธาตุอาวุธค้างใน SQL ไม่ให้ apply bAtkEle อีก
	.@s$ = replacestr(.@s$, "bonus bAtkEle,1; ", "");
	.@s$ = replacestr(.@s$, "bonus bAtkEle,2; ", "");
	.@s$ = replacestr(.@s$, "bonus bAtkEle,3; ", "");
	.@s$ = replacestr(.@s$, "bonus bAtkEle,4; ", "");
	.@s$ = replacestr(.@s$, "bonus bAtkEle,6; ", "");
	if (substr(.@s$, 0, 1) != "{") .@s$ = "{ " + .@s$;
	.@len = getstrlen(.@s$);
	if (.@len > 0 && substr(.@s$, .@len-1, 1) != "}") .@s$ = .@s$ + " }";
	return .@s$;
}

function	script	F_IBUFF_SyncAndRebuild_SlotBound	{
	.@cid = getcharid(3);
	.@now = gettimetick(2);
	query_sql("DELETE FROM `item_buff_slot` WHERE `expire_ts` <= "+.@now);
	
	// ล้างบัฟเก่า (ปลอดภัยกับกั้ม ใช้ Flag 2)
	bonus_script_clear 2;
	
	deltimer "ItemBuffSystem::OnIBUFF_ExpireTick"; 
	deltimer "ItemBuffSystem::OnIBUFF_Warn10Min"; 
	deltimer "ItemBuffSystem::OnIBUFF_Warn1Min";
	
	.@next_exp = 0; .@next_idx = -1; 
	
	for (.@i=0; .@i<6; .@i++) {
		.@slot = callfunc("F_IBUFF_IdxToSlot", .@i); 
		.@eqid = getequipid(.@slot);
		
		if (.@eqid < 0) { 
			query_sql("DELETE FROM `item_buff_slot` WHERE `char_id`="+.@cid+" AND `slot_idx`="+.@i+" AND `is_locked`=0");
			continue; 
		}
		
		.@eqref = getequiprefinerycnt(.@slot);
		.@r = query_sql("SELECT `item_id`,`refine`,`script`,`expire_ts`,`is_locked` FROM `item_buff_slot` WHERE `char_id`="+.@cid+" AND `slot_idx`="+.@i+" LIMIT 1", .@db_iid, .@db_ref, .@scr$, .@exp, .@locked);
		if (.@r <= 0) continue;
		
		if (.@db_iid != .@eqid || .@db_ref != .@eqref) { 
			if (.@locked == 0) { query_sql("DELETE FROM `item_buff_slot` WHERE `char_id`="+.@cid+" AND `slot_idx`="+.@i); }
			continue; 
		}
		
		.@rem = .@exp - .@now;
		if (.@rem <= 0) { query_sql("DELETE FROM `item_buff_slot` WHERE `char_id`="+.@cid+" AND `slot_idx`="+.@i); continue; }
		
		// กำหนดไอคอนให้แสดงผลถูกต้อง (ใช้ EFST_ แทน SC_)
		.@icon = EFST_BLESSING;
		if (.@i == 0) .@icon = EFST_DEFENDER;
		else if (.@i == 1) .@icon = EFST_MAGNIFICAT;
		else if (.@i == 2) .@icon = EFST_REFLECTSHIELD;
		else if (.@i == 3) .@icon = EFST_WINDWALK;
		else if (.@i == 4) .@icon = EFST_ASSUMPTIO;
		else if (.@i == 5) .@icon = 0; // v4: ไม่มีสุ่มธาตุแล้ว ไม่ต้องโชว์ icon ธาตุ

		// แปะบัฟ (เวลาใส่เป็น .@rem วินาทีเพียวๆ ไม่ต้องคูณ 1000 แล้วครับ)
		.@tmp$ = callfunc("F_IBUFF_NormalizeScript$", .@scr$);
		bonus_script .@tmp$, .@rem, 2, 1, -1;
		
		if (.@next_exp == 0 || .@exp < .@next_exp) { .@next_exp = .@exp; .@next_idx = .@i; }
	}
	
	if (.@next_exp > 0) {
		.@sec = .@next_exp - .@now; if (.@sec < 1) .@sec = 1;
		@IBUFF_WARN_IDX = .@next_idx;
		if (.@sec > 600) addtimer ((.@sec - 600) * 1000), "ItemBuffSystem::OnIBUFF_Warn10Min";
		if (.@sec > 60) addtimer ((.@sec - 60) * 1000), "ItemBuffSystem::OnIBUFF_Warn1Min";
		addtimer (.@sec * 1000) + 50, "ItemBuffSystem::OnIBUFF_ExpireTick";
	} else { 
		@IBUFF_WARN_IDX = -1; 
	}
	return;
}
