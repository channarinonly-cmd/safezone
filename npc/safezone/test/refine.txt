function	script	F_SF_PosName	{
	.@pos = getarg(0);
	if (.@pos == EQI_HEAD_TOP)  return "Head Top";
	if (.@pos == EQI_HEAD_MID)  return "Head Mid";
	if (.@pos == EQI_HEAD_LOW)  return "Head Low";
	if (.@pos == EQI_ARMOR)     return "Armor";
	if (.@pos == EQI_HAND_L)    return "Left Hand";
	if (.@pos == EQI_HAND_R)    return "Right Hand";
	if (.@pos == EQI_GARMENT)   return "Garment";
	if (.@pos == EQI_SHOES)     return "Shoes";
	if (.@pos == EQI_ACC_L)     return "Accessory L";
	if (.@pos == EQI_ACC_R)     return "Accessory R";
	return "Unknown";
}

function	script	F_SF_XXNeed	{
	.@from = getarg(0); .@to = getarg(1);
	if (.@from == 0 && .@to == 1) return 5;
	if (.@from == 1 && .@to == 2) return 10;
	if (.@from == 2 && .@to == 3) return 15;
	if (.@from == 3 && .@to == 4) return 20;
	if (.@from == 4 && .@to == 5) return 25;
	if (.@from == 5 && .@to == 6) return 30;
	if (.@from == 6 && .@to == 7) return 35;
	if (.@from == 7 && .@to == 8) return 40;
	if (.@from == 8 && .@to == 9) return 45;
	if (.@from == 9 && .@to == 10) return 50;
	return 0; 
}

function	script	F_SF_ProtectNeed	{
	.@to = getarg(0);
	if (.@to == 5)  return 1; if (.@to == 6)  return 2;
	if (.@to == 7)  return 4; if (.@to == 8)  return 7;
	if (.@to == 9)  return 11; if (.@to == 10) return 16;
	return 0;
}

function	script	F_SF_SureNeed	{
	.@to = getarg(0);
	if (.@to <= 5)  return 1; if (.@to == 6)  return 2;
	if (.@to == 7)  return 3; if (.@to == 8)  return 4;
	if (.@to == 9)  return 5; if (.@to == 10) return 6;
	return 0;
}

-	script	SF_Refine#single	-1,{
	disable_items;

	.@npc_name$ = "^000080Upgrader^000000";
	.@menu$ = "";
	.@equipped = 0;

	setarray .@indices[0], EQI_HEAD_TOP, EQI_ARMOR, EQI_HAND_L, EQI_HAND_R, EQI_GARMENT, EQI_SHOES, EQI_ACC_L, EQI_ACC_R, EQI_HEAD_MID, EQI_HEAD_LOW;
	setarray .@valid_parts[0], 0;
	.@count = 0;

	for(.@i = 0; .@i < getarraysize(.@indices); ++.@i){
		if(getequipisequiped(.@indices[.@i])){
			.@menu$ += "^000080" + callfunc("F_SF_PosName", .@indices[.@i]) + "-[" + getequipname(.@indices[.@i]) + "]^000000:";
			.@valid_parts[.@count] = .@indices[.@i];
			.@count++;
			.@equipped = 1;
		}
	}

	if(!.@equipped){
		mes "["+.@npc_name$+"]"; mes "ไม่พบไอเท็มที่สวมใส่"; goto L_Close;
	}

	.@menu$ += "^FF0000-- ยกเลิก (Cancel) --^000000";
	mes "["+.@npc_name$+"]"; mes "เลือกชิ้นส่วนที่ต้องการตีบวก:";
	
	.@sel = select(.@menu$);
	if (.@sel == 255) goto L_ForceEnd;
	
	.@s = .@sel - 1;
	if (.@s >= .@count) goto L_Close;
	
	.@part = .@valid_parts[.@s];

	if(!getequipisenableref(.@part)){
		mes "["+.@npc_name$+"]"; mes "ไอเท็มนี้ไม่สามารถตีบวกได้"; goto L_Close;
	}

	.@refine = getequiprefinerycnt(.@part);
	if(.@refine >= .MaxRefine){
		mes "["+.@npc_name$+"]"; mes "ไอเท็มนี้บวกสูงสุดแล้ว"; goto L_Close;
	}

	.@target = .@refine + 1;
	.@itemid = getequipid(.@part);
	setarray .@card[0], getequipcardid(.@part,0), getequipcardid(.@part,1), getequipcardid(.@part,2), getequipcardid(.@part,3);

	// ====================================================
	// ดึงค่าจาก refine_db.yml โดยใช้ Constants
	// ====================================================
	
	// ดึงค่าสำหรับ Normal (หากไม่ได้ค่า จะติด -1)
	.@norm_mat   = getequiprefinecost(.@part, 0, REFINE_MATERIAL_ID);
	.@norm_price = getequiprefinecost(.@part, 0, REFINE_ZENY_COST);
	.@norm_rate  = getequippercentrefinery(.@part, 0); 

	// ดึงค่าสำหรับ Enriched
	.@enrich_mat   = getequiprefinecost(.@part, 2, REFINE_MATERIAL_ID);
	.@enrich_price = getequiprefinecost(.@part, 2, REFINE_ZENY_COST);
	.@enrich_rate  = getequippercentrefinery(.@part, 2); 

	// [Safety Check 1]: ตรวจสอบว่ามีข้อมูล Normal ใน Database หรือไม่
	if (.@norm_mat <= 0) {
		mes "["+.@npc_name$+"]";
		mes "^FF0000เกิดข้อผิดพลาด: ไม่พบข้อมูลแร่ตีบวกในฐานข้อมูล!^000000";
		mes "โปรดตรวจสอบไฟล์ refine_db.yml ว่าได้ตั้งค่าแร่ Normal สำหรับไอเทมระดับนี้ไว้แล้วหรือไม่";
		goto L_Close;
	}

	mes "["+.@npc_name$+"]";
	mes "ระดับ: ^0000FF+" + .@refine + "^000000 -> +" + .@target;
	if (.@target <= 4) mes "^FF0000(คำเตือน: หากพลาดในระดับนี้ ไอเทมจะแตกสลายทันที!)^000000";
	mes "กรุณาเลือกประเภทการตีบวกที่ต้องการ:";
	mes "^0088001. แบบ Normal^000000 (โอกาส " + .@norm_rate + "%)";
	
	if (.@enrich_mat > 0) {
		mes "^0000FF2. แบบ Enriched^000000 (โอกาส " + .@enrich_rate + "%)";
	}
	next;

	if (.@enrich_mat > 0) {
		.@mat_sel = select(
			"แบบ Normal - ใช้ " + getitemname(.@norm_mat) + " (เรท " + .@norm_rate + "%)",
			"แบบ Enriched - ใช้ " + getitemname(.@enrich_mat) + " (เรท " + .@enrich_rate + "%)",
			"ยกเลิก"
		);
	} else {
		.@mat_sel = select(
			"แบบ Normal - ใช้ " + getitemname(.@norm_mat) + " (เรท " + .@norm_rate + "%)",
			"ยกเลิก"
		);
		if (.@mat_sel == 2) .@mat_sel = 3; 
	}

	if (.@mat_sel == 3) goto L_Close;

	// รับค่าตามที่ผู้เล่นเลือก
	if (.@mat_sel == 1) {
		.@material = .@norm_mat;
		.@price    = .@norm_price;
		.@rate     = .@norm_rate;
	} else {
		.@material = .@enrich_mat;
		.@price    = .@enrich_price;
		.@rate     = .@enrich_rate;
	}

	// [Safety Check 2]: ป้องกันตัวแปร Material รั่วไหลก่อนเรียกใช้ countitem
	if (.@material <= 0) {
		mes "["+.@npc_name$+"]";
		mes "^FF0000เกิดข้อผิดพลาด: ID ของแร่ตีบวกไม่ถูกต้อง!^000000";
		goto L_Close;
	}

	.@hasSure = 0; .@sureNeed = 0;
	if (.SureItemID > 0) {
		.@sureNeed = callfunc("F_SF_SureNeed", .@target);
		if (.@sureNeed > 0) .@hasSure = 1;
	}

	.@hasProt = 0; .@protNeed = 0;
	if (.ProtectItemID > 0) {
		.@protNeed = callfunc("F_SF_ProtectNeed", .@target);
		if (.@protNeed > 0 && .@target >= .ProtectMinTarget) .@hasProt = 1; 
	}

	.@hasDownProt = 0; .@downProtNeed = 0;
	if (.DownProtItemID > 0 && .@target >= .DownProtMinTarget) {
		.@downProtNeed = .DownProtAmount;
		.@hasDownProt = 1; 
	}

	.@hasXX = 0; .@xxNeed = callfunc("F_SF_XXNeed", .@refine, .@target);
	if (.XXItemID > 0) {
		if (.@xxNeed > 0) {
			if (countitem(.XXItemID) >= .@xxNeed) .@hasXX = 1;
		}
	}

	mes "["+.@npc_name$+"]";
	mes "ไอเท็ม: ^FF0000" + getequipname(.@part) + "^000000";
	mes "ระดับ: ^0000FF+" + .@refine + "^000000 -> +" + .@target;
	mes "โอกาสสำเร็จ: ^FF0000" + .@rate + "%^000000";
	mes "ใช้ Zeny: " + .@price;
	mes "ใช้วัตถุดิบ: " + getitemname(.@material) + " x1 (มี " + countitem(.@material) + ")";
	
	if (.XXItemID > 0 && .@xxNeed > 0) {
		mes "- " + getitemname(.XXItemID) + " x" + .@xxNeed + " (มี " + countitem(.XXItemID) + ")";
	}
	
	if ((.SureItemID > 0 && .@sureNeed > 0) || (.ProtectItemID > 0 && .@protNeed > 0 && .@target >= .ProtectMinTarget) || (.DownProtItemID > 0 && .@downProtNeed > 0 && .@target >= .DownProtMinTarget)) {
		mes "^0000FF[ ออปชั่นเสริม ]^000000";
		if (.SureItemID > 0 && .@sureNeed > 0) mes "- การันตีติด: ใช้ x" + .@sureNeed + " (มี " + countitem(.SureItemID) + ")";
		if (.ProtectItemID > 0 && .@protNeed > 0 && .@target >= .ProtectMinTarget) mes "- กันแตก (ลด 1 ขั้น): ใช้ x" + .@protNeed + " (มี " + countitem(.ProtectItemID) + ")";
		if (.DownProtItemID > 0 && .@downProtNeed > 0 && .@target >= .DownProtMinTarget) mes "- ใบกันลดขั้น (ต้องใช้คู่กันแตก): ใช้ x" + .@downProtNeed + " (มี " + countitem(.DownProtItemID) + ")";
	}
	next;

	.@useSure = 0; .@useProt = 0; .@useDownProt = 0;
	
	if (.@hasSure) {
		if (select("ไม่ใช้การันตีติด:ใช้การันตีติด [x"+.@sureNeed+"]") == 2) .@useSure = 1;
	}
	
	if (!.@useSure) {
		if (.@hasProt && .@hasDownProt) {
			.@ans2 = select(
				"ไม่ใช้อุปกรณ์ป้องกันใดๆ (เสี่ยงแตก)",
				"ใช้เฉพาะ [กันแตก] (พลาด=ลด 1 ขั้น) [ใช้ x"+.@protNeed+"]",
				"ใช้ [กันแตก] + [ใบกันลด] (พลาด=ไม่ลดขั้น) [ใช้ x"+.@downProtNeed+"]"
			);
			if (.@ans2 == 255) goto L_ForceEnd;
			if (.@ans2 == 2) .@useProt = 1;
			if (.@ans2 == 3) { .@useProt = 1; .@useDownProt = 1; }
		} else if (.@hasProt) {
			.@ans2 = select("ไม่ใช้อุปกรณ์ป้องกันใดๆ (เสี่ยงแตก):ใช้กันแตก (ล้มเหลวลด 1 ขั้น) [ใช้ x"+.@protNeed+"]");
			if (.@ans2 == 255) goto L_ForceEnd;
			if (.@ans2 == 2) .@useProt = 1;
		}
	}

	if (.@protNeed > 0 && !.@useProt && !.@useSure) {
		mes "^FF0000[ คำเตือน ] คุณไม่ได้ใช้อุปกรณ์ป้องกันใดๆ หากพลาดไอเทมอาจแตกสลายได้!^000000";
	}
	
	mes "["+.@npc_name$+"]";
	mes "ตรวจสอบข้อมูลให้เรียบร้อย...";
	if(select("ยืนยันตีบวก:ยกเลิก") == 2) goto L_Close;

	if (!getequipisequiped(.@part) || getequipid(.@part) != .@itemid || getequiprefinerycnt(.@part) != .@refine) {
		mes "["+.@npc_name$+"]"; mes "^FF0000ตรวจพบการเปลี่ยนไอเท็ม!^000000"; goto L_Close;
	}

	if (Zeny < .@price) { mes "["+.@npc_name$+"]"; mes "^FF0000Zeny ไม่พอ^000000"; goto L_Close; }
	
	// [Safety Check 3]: ป้องกันการตัดแร่ผิดพลาด
	if (.@material > 0) {
		if (countitem(.@material) < 1) { mes "["+.@npc_name$+"]"; mes "^FF0000วัตถุดิบตีบวกไม่พอ^000000"; goto L_Close; }
	} else {
		mes "["+.@npc_name$+"]"; mes "^FF0000เกิดข้อผิดพลาดในการโหลดไอเทม!^000000"; goto L_Close;
	}
	
	if (.XXItemID > 0 && .@xxNeed > 0) {
		if (countitem(.XXItemID) < .@xxNeed) { mes "["+.@npc_name$+"]"; mes "^FF0000ไอเท็มพิเศษไม่พอ^000000"; goto L_Close; }
	}
	if (.@useSure) {
		if (countitem(.SureItemID) < .@sureNeed) { mes "["+.@npc_name$+"]"; mes "^FF0000ไอเท็มการันตีไม่พอ^000000"; goto L_Close; }
	}
	if (.@useProt) {
		if (countitem(.ProtectItemID) < .@protNeed) { mes "["+.@npc_name$+"]"; mes "^FF0000ไอเท็มกันแตกไม่พอ^000000"; goto L_Close; }
	}
	if (.@useDownProt) {
		if (countitem(.DownProtItemID) < .@downProtNeed) { mes "["+.@npc_name$+"]"; mes "^FF0000ใบกันลดขั้นไม่พอ^000000"; goto L_Close; }
	}

	if (.@material > 0) delitem .@material, 1;
	Zeny -= .@price;
	if (.XXItemID > 0 && .@xxNeed > 0) delitem .XXItemID, .@xxNeed;
	if (.@useSure) delitem .SureItemID, .@sureNeed;
	if (.@useProt) delitem .ProtectItemID, .@protNeed;
	if (.@useDownProt) delitem .DownProtItemID, .@downProtNeed;

	// ==========================================
	// ประมวลผลโอกาสตีบวก (Rigged System / RNG)
	// ==========================================
	.@isSuccess = 0;

	if (.@useSure) {
		.@isSuccess = 1;
	} 
	else if (!.@useDownProt) {
		if (rand(100) < .@rate) {
			.@isSuccess = 1;
			SF_FailCount = 0; SF_TargetFails = 0;
		}
	} 
	else {
		if (.EnableRiggedSystem) {
			if (SF_TargetFails == 0) SF_TargetFails = rand(.MinForceFail, .MaxForceFail);
	
			if (SF_FailCount < SF_TargetFails) {
				SF_FailCount++; 
				.@isSuccess = 0;
			} else {
				if (rand(100) < .@rate) {
					.@isSuccess = 1;
					SF_FailCount = 0; SF_TargetFails = 0;
				} else {
					SF_FailCount++; 
					.@isSuccess = 0;
				}
			}
		} else {
			if (rand(100) < .@rate) {
				.@isSuccess = 1;
			}
		}
	}

	// ==========================================
	// ประกาศผล
	// ==========================================
	if (.@isSuccess) {
		successrefitem .@part;
		specialeffect2 153; emotion ET_BEST;
		mes "["+.@npc_name$+"]"; mes "^00FF00สำเร็จ!^000000 ตอนนี้เป็น +" + getequiprefinerycnt(.@part);

		if (.@target >= 7) {
			announce "ยินดีด้วย! ผู้เล่น ["+strcharinfo(0)+"] ตีบวก "+getitemname(.@itemid)+" สำเร็จเป็น +"+.@target+"!", bc_all|bc_blue;
			query_sql "INSERT INTO `log_refine_success` (`char_name`, `item_name`, `refine_level`) VALUES ('"+escape_sql(strcharinfo(0))+"', '"+escape_sql(getitemname(.@itemid))+"', "+.@target+")";
		}
		goto L_Close;
	}

	// ==========================================
	// ระบบล้มเหลว (บังคับแตก +1 ถึง +4)
	// ==========================================
	specialeffect2 EF_REFINE_FAIL; emotion ET_HUK;

	if (.@target <= 4) {
		failedrefitem .@part; 
		mes "["+.@npc_name$+"]"; mes "^FF0000ตีบวกไม่สำเร็จ ไอเทมแตกสลาย!^000000";
		goto L_Close;
	}

	if (.@useProt && .@useDownProt) {
		mes "["+.@npc_name$+"]"; mes "^FF0000ตีบวกไม่สำเร็จ^000000 (ใบกันลดขั้นทำงาน) ระดับไอเทมเท่าเดิม";
		goto L_Close;
	}

	if (.@useProt) {
		if (getequiprefinerycnt(.@part) > 0) downrefitem .@part;
		mes "["+.@npc_name$+"]"; mes "^FF0000ตีบวกไม่สำเร็จ^000000 (กันแตกทำงาน) ลดระดับลง 1 ขั้น";
		goto L_Close;
	}

	if (.FailMode == 0) {
		if (.FailBreakChance > 0 && .@refine >= .FailBreakFromRefine && rand(100) < .FailBreakChance) {
			failedrefitem .@part;
			mes "["+.@npc_name$+"]"; mes "^FF0000ตีบวกแตกสลาย!^000000";
			goto L_Close;
		}
		if (getequiprefinerycnt(.@part) > 0) downrefitem .@part;
		mes "["+.@npc_name$+"]"; mes "^FF0000ล้มเหลว!^000000 ลดระดับลง 1 ขั้น";
		goto L_Close;
	}

	failedrefitem .@part;
	mes "["+.@npc_name$+"]"; mes "^FF0000ตีบวกแตกสลาย!^000000";

L_Close:
	enable_items; close;
	
L_ForceEnd:
	enable_items; end;

OnInit:
	waitingroom "ตีบวกไอเทม", 0; 
	.EnableRiggedSystem = 0;
	.MinForceFail = 2;
	.MaxForceFail = 5;
	.XXItemID = 6635;  
	.SureItemID = 0;   
	.MaxRefine = 10;
	.FailMode = 0;
	.FailBreakFromRefine = 4;
	.FailBreakChance = 100;
	.ProtectItemID = 6634; 
	.ProtectMinTarget = 7;
	.DownProtItemID = 6633; 
	.DownProtMinTarget = 8; 
	.DownProtAmount = 1; 
end;
}

//===========================================================

function	script	F_Shadow_PosName	{
	.@pos = getarg(0);
	if (.@pos == EQI_SHADOW_ARMOR)  return "Shadow Armor";
	if (.@pos == EQI_SHADOW_SHIELD) return "Shadow Shield";
	if (.@pos == EQI_SHADOW_SHOES)  return "Shadow Shoes";
	if (.@pos == EQI_SHADOW_ACC_R)  return "Shadow Earring"; 
	if (.@pos == EQI_SHADOW_ACC_L)  return "Shadow Pendant"; 
	return "Unknown";
}

function	script	F_Shadow_XXNeed	{
	.@from = getarg(0); .@to = getarg(1);
	if (.@from == 0 && .@to == 1) return 5;
	if (.@from == 1 && .@to == 2) return 10;
	if (.@from == 2 && .@to == 3) return 15;
	if (.@from == 3 && .@to == 4) return 20;
	if (.@from == 4 && .@to == 5) return 25;
	if (.@from == 5 && .@to == 6) return 30;
	if (.@from == 6 && .@to == 7) return 35;
	if (.@from == 7 && .@to == 8) return 40;
	if (.@from == 8 && .@to == 9) return 45;
	if (.@from == 9 && .@to == 10) return 50;
	return 0; 
}

	if (.@to == 5)  return 1; if (.@to == 6)  return 2;
	if (.@to == 7)  return 4; if (.@to == 8)  return 7;
	if (.@to == 9)  return 11; if (.@to == 10) return 16;
	return 0;
}

function	script	F_Shadow_SureNeed	{

	if (.@to <= 5)  return 1; if (.@to == 6)  return 2;
	if (.@to == 7)  return 3; if (.@to == 8)  return 4;
	if (.@to == 9)  return 5; if (.@to == 10) return 6;
	return 0;
}

-	script	Shadow_Refine#single	-1,{
	disable_items;

	.@npc_name$ = "^000080Shadow Artisan^000000";
	.@menu$ = "";
	.@equipped = 0;

	setarray .@indices[0], EQI_SHADOW_ARMOR, EQI_SHADOW_SHIELD, EQI_SHADOW_SHOES, EQI_SHADOW_ACC_R, EQI_SHADOW_ACC_L;
	setarray .@valid_parts[0], 0;
	.@count = 0;

	for(.@i = 0; .@i < getarraysize(.@indices); ++.@i){
		if(getequipisequiped(.@indices[.@i])){
			.@menu$ += "^000080" + callfunc("F_Shadow_PosName", .@indices[.@i]) + "-[" + getequipname(.@indices[.@i]) + "]^000000:";
			.@valid_parts[.@count] = .@indices[.@i];
			.@count++;
			.@equipped = 1;
		}
	}

	if(!.@equipped){
		mes "["+.@npc_name$+"]"; mes "ไม่พบ Shadow Equipment ที่สวมใส่"; goto L_Close;
	}

	.@menu$ += "^FF0000-- ยกเลิก (Cancel) --^000000";
	mes "["+.@npc_name$+"]"; mes "เลือกชิ้นส่วน Shadow ที่ต้องการตีบวก:";
	
	.@sel = select(.@menu$);
	if (.@sel == 255) goto L_ForceEnd;
	
	.@s = .@sel - 1;
	if (.@s >= .@count) goto L_Close;
	
	.@part = .@valid_parts[.@s];

	if(!getequipisenableref(.@part)){
		mes "["+.@npc_name$+"]"; mes "ไอเท็มนี้ไม่สามารถตีบวกได้"; goto L_Close;
	}

	.@refine = getequiprefinerycnt(.@part);
	if(.@refine >= .MaxRefine){
		mes "["+.@npc_name$+"]"; mes "ไอเท็มนี้บวกสูงสุดแล้ว"; goto L_Close;
	}

	.@target = .@refine + 1;
	.@itemid = getequipid(.@part);

	// ====================================================
	// ดึงค่าจาก refine_db.yml อย่างถูกต้อง 100%
	// ====================================================
	
	// ดึงค่าสำหรับ Normal (Refine Type = 0)
	.@norm_mat   = getequiprefinecost(.@part, 1, 0); 
	.@norm_price = getequiprefinecost(.@part, 0, 0); 
	.@norm_rate  = getequippercentrefinery(.@part, 0); 

	// ดึงค่าสำหรับ Enriched (Refine Type = 1)
	.@enrich_mat   = getequiprefinecost(.@part, 1, 1); 
	.@enrich_price = getequiprefinecost(.@part, 0, 1); 
	.@enrich_rate  = getequippercentrefinery(.@part, 1); 

	if (.@norm_mat <= 0) {
		mes "["+.@npc_name$+"]";
		mes "^FF0000เกิดข้อผิดพลาด: ไม่พบข้อมูลแร่ Shadow ในฐานข้อมูล!^000000";
		goto L_Close;
	}

	mes "["+.@npc_name$+"]";
	mes "ระดับ: ^0000FF+" + .@refine + "^000000 -> +" + .@target;
	if (.@target <= 4) mes "^FF0000(คำเตือน: หากพลาดในระดับนี้ ไอเทมจะแตกสลายทันที!)^000000";
	mes "กรุณาเลือกประเภทการตีบวกที่ต้องการ:";
	mes "^0088001. แบบ Normal^000000 (โอกาส " + .@norm_rate + "%)";
	
	if (.@enrich_mat > 0) {
		mes "^0000FF2. แบบ Enriched^000000 (โอกาส " + .@enrich_rate + "%)";
	}
	next;

	if (.@enrich_mat > 0) {
		.@mat_sel = select(
			"แบบ Normal - ใช้ " + getitemname(.@norm_mat) + " (เรท " + .@norm_rate + "%)",
			"แบบ Enriched - ใช้ " + getitemname(.@enrich_mat) + " (เรท " + .@enrich_rate + "%)",
			"ยกเลิก"
		);
	} else {
		.@mat_sel = select(
			"แบบ Normal - ใช้ " + getitemname(.@norm_mat) + " (เรท " + .@norm_rate + "%)",
			"ยกเลิก"
		);
		if (.@mat_sel == 2) .@mat_sel = 3; 
	}

	if (.@mat_sel == 3) goto L_Close;

	// รับค่าตามที่ผู้เล่นเลือก
	if (.@mat_sel == 1) {
		.@material = .@norm_mat;
		.@price    = .@norm_price;
		.@rate     = .@norm_rate;
	} else {
		.@material = .@enrich_mat;
		.@price    = .@enrich_price;
		.@rate     = .@enrich_rate;
	}

	if (.@material <= 0) {
		mes "["+.@npc_name$+"]";
		mes "^FF0000เกิดข้อผิดพลาด: ID ของแร่ตีบวกไม่ถูกต้อง!^000000";
		goto L_Close;
	}

	.@hasSure = 0; .@sureNeed = 0;
	if (.SureItemID > 0) {
		.@sureNeed = callfunc("F_Shadow_SureNeed", .@target);
		if (.@sureNeed > 0) .@hasSure = 1;
	}

	.@hasProt = 0; .@protNeed = 0;
	if (.ProtectItemID > 0) {
		.@protNeed = callfunc("F_Shadow_ProtectNeed", .@target);
		if (.@protNeed > 0 && .@target >= .ProtectMinTarget) .@hasProt = 1; 
	}

	.@hasDownProt = 0; .@downProtNeed = 0;
	if (.DownProtItemID > 0 && .@target >= .DownProtMinTarget) {
		.@downProtNeed = .DownProtAmount;
		.@hasDownProt = 1; 
	}

	.@hasXX = 0; .@xxNeed = callfunc("F_Shadow_XXNeed", .@refine, .@target);
	if (.XXItemID > 0) {
		if (.@xxNeed > 0) {
			if (countitem(.XXItemID) >= .@xxNeed) .@hasXX = 1;
		}
	}

	mes "["+.@npc_name$+"]";
	mes "ไอเท็ม: ^FF0000" + getequipname(.@part) + "^000000";
	mes "ระดับ: ^0000FF+" + .@refine + "^000000 -> +" + .@target;
	mes "โอกาสสำเร็จ: ^FF0000" + .@rate + "%^000000";
	mes "ใช้ Zeny: ^0000FF" + callfunc("F_InsertComma", .@price) + "^000000";
	mes "ใช้วัตถุดิบ: " + getitemname(.@material) + " x1 (มี " + countitem(.@material) + ")";

	if (.XXItemID > 0 && .@xxNeed > 0) {
		mes "- " + getitemname(.XXItemID) + " x" + .@xxNeed + " (มี " + countitem(.XXItemID) + ")";
	}
	
	if ((.SureItemID > 0 && .@sureNeed > 0) || (.ProtectItemID > 0 && .@protNeed > 0 && .@target >= .ProtectMinTarget) || (.DownProtItemID > 0 && .@downProtNeed > 0 && .@target >= .DownProtMinTarget)) {
		mes "^0000FF[ ออปชั่นเสริม ]^000000";
		if (.SureItemID > 0 && .@sureNeed > 0) mes "- การันตีติด: ใช้ x" + .@sureNeed + " (มี " + countitem(.SureItemID) + ")";
		if (.ProtectItemID > 0 && .@protNeed > 0 && .@target >= .ProtectMinTarget) mes "- กันแตก (ลด 1 ขั้น): ใช้ x" + .@protNeed + " (มี " + countitem(.ProtectItemID) + ")";
		if (.DownProtItemID > 0 && .@downProtNeed > 0 && .@target >= .DownProtMinTarget) mes "- ใบกันลดขั้น (ต้องใช้คู่กันแตก): ใช้ x" + .@downProtNeed + " (มี " + countitem(.DownProtItemID) + ")";
	}
	next;

	.@useSure = 0; .@useProt = 0; .@useDownProt = 0;
	
	if (.@hasSure) {
		if (select("ไม่ใช้การันตีติด:ใช้การันตีติด [x"+.@sureNeed+"]") == 2) .@useSure = 1;
	}
	
	if (!.@useSure) {
		if (.@hasProt && .@hasDownProt) {
			.@ans2 = select(
				"ไม่ใช้อุปกรณ์ป้องกันใดๆ (เสี่ยงแตก)",
				"ใช้เฉพาะ [กันแตก] (พลาด=ลด 1 ขั้น) [ใช้ x"+.@protNeed+"]",
				"ใช้ [กันแตก] + [ใบกันลด] (พลาด=ไม่ลดขั้น) [ใช้ x"+.@downProtNeed+"]"
			);
			if (.@ans2 == 255) goto L_ForceEnd;
			if (.@ans2 == 2) .@useProt = 1;
			if (.@ans2 == 3) { .@useProt = 1; .@useDownProt = 1; }
		} else if (.@hasProt) {
			.@ans2 = select("ไม่ใช้อุปกรณ์ป้องกันใดๆ (เสี่ยงแตก):ใช้กันแตก (ล้มเหลวลด 1 ขั้น) [ใช้ x"+.@protNeed+"]");
			if (.@ans2 == 255) goto L_ForceEnd;
			if (.@ans2 == 2) .@useProt = 1;
		}
	}

	if (.@protNeed > 0 && !.@useProt && !.@useSure) {
		mes "^FF0000[ คำเตือน ] คุณไม่ได้ใช้อุปกรณ์ป้องกันใดๆ หากพลาดไอเทมอาจแตกสลายได้!^000000";

	mes "["+.@npc_name$+"]";
	mes "ตรวจสอบข้อมูลให้เรียบร้อย...";
	if(select("ยืนยันตีบวก:ยกเลิก") == 2) goto L_Close;

	if (!getequipisequiped(.@part) || getequipid(.@part) != .@itemid || getequiprefinerycnt(.@part) != .@refine) {
		mes "["+.@npc_name$+"]"; mes "^FF0000ตรวจพบการเปลี่ยนไอเท็ม!^000000"; goto L_Close;
	}

	if (Zeny < .@price) { mes "["+.@npc_name$+"]"; mes "^FF0000Zeny ไม่พอ^000000"; goto L_Close; }
	
	if (.@material > 0) {
		if (countitem(.@material) < 1) { mes "["+.@npc_name$+"]"; mes "^FF0000วัตถุดิบตีบวกไม่พอ^000000"; goto L_Close; }
	}

		if (countitem(.XXItemID) < .@xxNeed) { mes "["+.@npc_name$+"]"; mes "^FF0000ไอเท็มพิเศษไม่พอ^000000"; goto L_Close; }
	}

		if (countitem(.SureItemID) < .@sureNeed) { mes "["+.@npc_name$+"]"; mes "^FF0000ไอเท็มการันตีไม่พอ^000000"; goto L_Close; }
	}
	if (.@useProt) {
		if (countitem(.ProtectItemID) < .@protNeed) { mes "["+.@npc_name$+"]"; mes "^FF0000ไอเท็มกันแตกไม่พอ^000000"; goto L_Close; }

		if (countitem(.DownProtItemID) < .@downProtNeed) { mes "["+.@npc_name$+"]"; mes "^FF0000ใบกันลดขั้นไม่พอ^000000"; goto L_Close; }
	}

	if (.@material > 0) delitem .@material, 1;
	Zeny -= .@price;
	if (.XXItemID > 0 && .@xxNeed > 0) delitem .XXItemID, .@xxNeed;
	if (.@useSure && .SureItemID > 0) delitem .SureItemID, .@sureNeed;
	if (.@useProt && .ProtectItemID > 0) delitem .ProtectItemID, .@protNeed;
	if (.@useDownProt && .DownProtItemID > 0) delitem .DownProtItemID, .@downProtNeed;

	// ==========================================
	// RNG Check & Rigged System
	// ==========================================
	.@isSuccess = 0;
	if (.@useSure) {
		.@isSuccess = 1;
	} else {
		if (.EnableRiggedSystem) {
			if (Shadow_TargetFails == 0) Shadow_TargetFails = rand(.MinForceFail, .MaxForceFail);
	
			if (Shadow_FailCount < Shadow_TargetFails) {
				Shadow_FailCount++; 
				.@isSuccess = 0; 
			} else {
				if (rand(100) < .@rate) {
					.@isSuccess = 1;
					Shadow_FailCount = 0; Shadow_TargetFails = 0; 
				} else {
					Shadow_FailCount++; 
					.@isSuccess = 0;
				}
			}
		} else {
			if (rand(100) < .@rate) {
				.@isSuccess = 1;
			}
		}
	}

	if (.@isSuccess) {
		successrefitem .@part;
		specialeffect2 153; emotion ET_BEST;
		mes "["+.@npc_name$+"]"; mes "^00FF00สำเร็จ!^000000 ตอนนี้เป็น +" + getequiprefinerycnt(.@part);

		if (.@target >= 7) {
			announce "ยินดีด้วย! ผู้เล่น ["+strcharinfo(0)+"] ตีบวก "+getitemname(.@itemid)+" สำเร็จเป็น +"+.@target+"!", bc_all|bc_blue;
			query_sql "INSERT INTO `log_refine_success` (`char_name`, `item_name`, `refine_level`) VALUES ('"+escape_sql(strcharinfo(0))+"', '"+escape_sql(getitemname(.@itemid))+"', "+.@target+")";
		}
		goto L_Close;
	}

	// ==========================================
	// ระบบล้มเหลว (บังคับแตก +1 ถึง +4)
	// ==========================================
	specialeffect2 EF_REFINE_FAIL; emotion ET_HUK;
	if (.@target <= 4) {
		failedrefitem .@part; 
		mes "["+.@npc_name$+"]"; mes "^FF0000ตีบวกไม่สำเร็จ ไอเทมแตกสลาย!^000000";
		goto L_Close;
	}
	if (.@useProt && .@useDownProt) {
		mes "["+.@npc_name$+"]"; mes "^FF0000ตีบวกไม่สำเร็จ^000000 (ใบกันลดทำงาน) ระดับเท่าเดิม";
		goto L_Close;
	}
	if (.@useProt) {
		if (getequiprefinerycnt(.@part) > 0) downrefitem .@part;
		mes "["+.@npc_name$+"]"; mes "^FF0000ตีบวกไม่สำเร็จ^000000 (กันแตกทำงาน) ลด 1 ขั้น";
		goto L_Close;
	}
	if (.FailMode == 0) {
		if (.FailBreakChance > 0 && .@refine >= .FailBreakFromRefine && rand(100) < .FailBreakChance) {
			failedrefitem .@part;
			mes "["+.@npc_name$+"]"; mes "^FF0000ตีบวกแตกสลาย!^000000";
			goto L_Close;
		}
		if (getequiprefinerycnt(.@part) > 0) downrefitem .@part;
		mes "["+.@npc_name$+"]"; mes "^FF0000ล้มเหลว!^000000 ลด 1 ขั้น";
		goto L_Close;
	}
	failedrefitem .@part;
	mes "["+.@npc_name$+"]"; mes "^FF0000ตีบวกแตกสลาย!^000000";

L_Close:
	enable_items; close;
L_ForceEnd:
	enable_items; end;

OnInit:
	waitingroom "ตีบวก Shadow", 0; 
	
	.EnableRiggedSystem = 0; 
	.MinForceFail = 2;   
	.MaxForceFail = 5;  
	.XXItemID = 6635;  
	.SureItemID = 0;   
	.MaxRefine = 10;
	.FailMode = 0;
	.FailBreakFromRefine = 4;
	.FailBreakChance = 100;
	
	.ProtectItemID = 6634;
	.ProtectMinTarget = 7;
	
	.DownProtItemID = 6633;
	.DownProtMinTarget = 8;
	.DownProtAmount = 1;
	
	// ตัดอาเรย์ Zeny, Rate ทิ้งให้หมด!
end;
}

		mes "^0000FF[ ทางเลือกเสริมสำหรับขั้น +" + .@target + " ]^000000";
		if (.ProtectItemID > 0 && .@protNeed > 0 && .@target >= .ProtectMinTarget) mes "- กันตีดรอป (ลด 1 ขั้น): ใช้ x" + .@protNeed + " (มี " + countitem(.ProtectItemID) + ")";
		if (.DownProtItemID > 0 && .@downProtNeed > 0 && .@target >= .DownProtMinTarget) mes "- ใบกันลดขั้น (ต้องใช้คู่กันแตก): ใช้ x" + .@downProtNeed + " (มี " + countitem(.DownProtItemID) + ")";
	}

	next;
	.@useProt = 0; .@useXX = 1; .@useDownProt = 0;

	if (.@hasProt && .@hasDownProt) {
		.@ans1 = select(
			"ไม่ใช้อุปกรณ์ป้องกันใดๆ (เสี่ยงแตก)",
			"ใช้เฉพาะ [กันแตก] (พลาด=ลด 1 ขั้น) [ใช้ x"+.@protNeed+"]",
			"ใช้ [กันแตก] + [ใบกันลด] (พลาด=ไม่ลดขั้น) [ใช้ x"+.@downProtNeed+"]"
		);
		if (.@ans1 == 255) goto L_Close;
		if (.@ans1 == 2) .@useProt = 1;
		if (.@ans1 == 3) { .@useProt = 1; .@useDownProt = 1; }
	} else if (.@hasProt) {
		.@ans1 = select("ไม่ใช้อุปกรณ์ป้องกันใดๆ (เสี่ยงแตก):ใช้กันแตก (ล้มเหลวลด 1 ขั้น) [ใช้ x"+.@protNeed+"]");
		if (.@ans1 == 255) goto L_Close;
		if (.@ans1 == 2) .@useProt = 1;
	}

	if (.@protNeed > 0 && !.@useProt) {
		mes "^FF0000[ คำเตือน ] คุณไม่ได้ใช้อุปกรณ์ป้องกัน หากพลาด Charm อาจแตกสลายได้!^000000";
	}
	
	mes "["+.@npc_name$+"]";
	mes "ตรวจสอบข้อมูลให้เรียบร้อย...";
	if(select("ยืนยันตีบวก:ยกเลิก") == 2) goto L_Close;

	getinventorylist;
	if (@inventorylist_id[.@idx] != .@itemid || @inventorylist_refine[.@idx] != .@refine) {
		mes "["+.@npc_name$+"]"; 
		mes "^FF0000ตรวจพบความผิดปกติ!^000000";
		mes "มีการเคลื่อนย้ายไอเทมระหว่างทำรายการ กรุณายกเลิกแล้วทำใหม่"; 
		goto L_Close;
	}

	if(Zeny < .ZenyCost){
		mes "["+.@npc_name$+"]"; mes "Zeny ไม่เพียงพอ"; goto L_Close;
	}
	if(.@material > 0 && countitem(.@material) < 1){
		mes "["+.@npc_name$+"]"; mes "วัตถุดิบไม่พอ"; goto L_Close;
	}

	if(.@useProt){
		if (.ProtectItemID <= 0) { mes "["+.@npc_name$+"]"; mes "ไอเทมกันแตกไม่ถูกตั้งค่า กรุณาติดต่อ GM"; goto L_Close; }
		if (.@protNeed > 0 && countitem(.ProtectItemID) < .@protNeed){
			mes "["+.@npc_name$+"]"; mes "^FF0000ไอเท็มกันแตกไม่พอ!^000000"; goto L_Close;
		}
	}
	if(.@useDownProt){
		if (.DownProtItemID <= 0) { mes "["+.@npc_name$+"]"; mes "ไอเทมกันลดขั้นไม่ถูกตั้งค่า กรุณาติดต่อ GM"; goto L_Close; }
		if (.@downProtNeed > 0 && countitem(.DownProtItemID) < .@downProtNeed){
			mes "["+.@npc_name$+"]"; mes "^FF0000ใบกันลดขั้นไม่พอ!^000000"; goto L_Close;
		}
	}
	if(.@xxNeed > 0){
		if (.XXItemID <= 0) { mes "[" + .@npc_name$ + "]"; mes "ไอเท็มพิเศษไม่ถูกตั้งค่า"; goto L_Close; }
		if (countitem(.XXItemID) < .@xxNeed){
			mes "[" + .@npc_name$ + "]"; mes "ไอเท็มพิเศษไม่พอ"; goto L_Close;
		}
	} 
		
	if(.@material > 0) delitem .@material, 1;
	Zeny -= .ZenyCost;
	
	if(.@useProt) delitem .ProtectItemID, .@protNeed;
	if(.@useDownProt) delitem .DownProtItemID, .@downProtNeed;
	if(.@useXX && .@xxNeed > 0) delitem .XXItemID, .@xxNeed; 

	if(rand(100) < .@rate){
		charm_setrefine .@itemid, .@target; 
		
		specialeffect2 153; 
		emotion ET_BEST;
		mes "["+.@npc_name$+"]"; mes "^00FF00สำเร็จ!^000000 ตอนนี้ Charm ของเจ้าเป็น +" + .@target;

		if (.@target >= 7) {
			announce "ยินดีด้วย! ผู้เล่น ["+strcharinfo(0)+"] ตีบวก "+getitemname(.@itemid)+" สำเร็จเป็น +"+.@target+"!", bc_all|bc_blue;
			query_sql "INSERT INTO `log_refine_success` (`char_name`, `item_name`, `refine_level`) VALUES ('"+escape_sql(strcharinfo(0))+"', '"+escape_sql(getitemname(.@itemid))+"', "+.@target+")";
		}

		goto L_Close; 
	}

	// ==========================================
	// ระบบล้มเหลว (บังคับแตก +1 ถึง +4)
	// ==========================================
	specialeffect2 EF_REFINE_FAIL;
	emotion ET_HUK;

	if (.@target <= 4) {
		delitem .@itemid, 1;
		mes "["+.@npc_name$+"]"; mes "^FF0000ตีบวกไม่สำเร็จ Charm แตกสลาย!^000000";
		goto L_Close;
	}

	if(.@useProt && .@useDownProt){
		mes "["+.@npc_name$+"]"; mes "^FF0000ตีบวกไม่สำเร็จ^000000 (ใบกันลดขั้นทำงาน) ระดับคงเดิม";
		goto L_Close;
	}

	if(.@useProt){
		.@new_refine = (.@refine > 0) ? (.@refine - 1) : 0;
		charm_setrefine .@itemid, .@new_refine; 
		mes "["+.@npc_name$+"]"; mes "^FF0000ตีบวกไม่สำเร็จ^000000 (กันแตกทำงาน) ลดระดับลง 1 ขั้น";
		goto L_Close;
	}

	if(.FailMode == 0){
		if(.FailBreakChance > 0 && .@refine >= .FailBreakFromRefine && rand(100) < .FailBreakChance){
			delitem .@itemid, 1; 
			mes "["+.@npc_name$+"]"; mes "^FF0000ตีบวกแตก!^000000 (สลายหายไป)";
			goto L_Close; 
		}
		.@new_refine = (.@refine > 0) ? (.@refine - 1) : 0;
		charm_setrefine .@itemid, .@new_refine; 
		mes "["+.@npc_name$+"]"; mes "^FF0000ตีบวกไม่สำเร็จ^000000 ลดระดับลง 1 ขั้น";
		goto L_Close;
	}

	delitem .@itemid, 1;
	mes "["+.@npc_name$+"]"; mes "^FF0000ตีบวกแตก!^000000"; goto L_Close;

L_Close:
	enable_items;
	close;

OnInit:
	waitingroom "อัปเกรด Charm", 0;
	.MaxRefine = 10;
	.ZenyCost = 10000;
	.XXItemID = 6635;  
	
	.ProtectItemID = 6634; 
	.ProtectMinTarget = 7;
	.DownProtItemID = 6633; 
	.DownProtMinTarget = 8; 
	.DownProtAmount = 1; 
	
	.FailMode = 0;
	.FailBreakFromRefine = 4;
	.FailBreakChance = 100;

	// ตั้งค่าแร่และเรทตีบวกของ Charm
	.NormMaterialID = 969;  // แร่ Normal ของ Charm
	.EnrichMaterialID = 7620; // แร่ Enriched ของ Charm (ถ้าไม่มีใส่ 0 ไว้เพื่อซ่อนเมนู)
	
	// ตั้งค่าเรท % โอกาสติด (ตัวอย่าง: ตำแหน่งที่ 1 คือ +0 ไป +1)
	setarray .NormRates[0], 0, 89, 89, 89, 89, 60, 40, 40, 20, 20, 9;
	setarray .EnrichRates[0], 0, 99, 99, 99, 99, 70, 50, 50, 30, 30, 15;
end;
}
