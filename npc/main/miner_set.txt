-	script	miner_status_clear	-1,{

OnPCLoginEvent:
	query_sql("DELETE FROM `tharakan_miner` WHERE `char_id` = '"+getcharid(0)+"'");
	undisguise;
	set minerstus,0;
end;

OnPCLogoutEvent:
	query_sql("DELETE FROM `tharakan_miner` WHERE `char_id` = '"+getcharid(0)+"'");
	undisguise;
	set minerstus,0;
end;

OnPCDieEvent:
	query_sql("DELETE FROM `tharakan_miner` WHERE `char_id` = '"+getcharid(0)+"'");
	undisguise;
	set minerstus,0;
end;

OnPCMoveEvent:
	query_sql("DELETE FROM `tharakan_miner` WHERE `char_id` = '"+getcharid(0)+"'");
	undisguise;
	set minerstus,0;
end;

}
